function prosesPembayaran(total, metode, isMember) {
  let diskon = 0;

  if (isMember) {
    if (total >= 100000) {
      diskon = 0.1 * total;
    } else {
      diskon = 0.05 * total;
    }
  }

  let totalBayar = total - diskon;

  if (metode === "cash") {
    return "Bayar tunai sebesar " + totalBayar;
  } else if (metode === "qris") {
    return "Scan QR untuk membayar " + totalBayar;
  } else {
    return "Metode pembayaran tidak dikenali";
  }
}
